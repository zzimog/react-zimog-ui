## Usage

```tsx
import { Button, Presence } from '@ui';

export default () => {
  const [present, setPresent] = useState(false);

  return (
    <>
      <Button onClick={() => setPresent(!present)} />
      <Presence
        data-present={present}
        present={present}
        className="data-[present=true]:animate-in data-[present=false]:animate-out"
      />
    </>
  );
};
```
