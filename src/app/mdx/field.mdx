# Field

Use `<Field />` component to compose form fields or grouped inputs, with labels, descriptions, and help texts.

<Demo>
  <div className="mx-auto flex max-w-lg flex-col gap-4">

      <Field.Label as="legend" children="Birthday" />
      <div className="grid grid-cols-3 gap-4">
        <Field>
          <Field.Label children="Day" />
          <Select>
            <Select.Option value="1" children="One" />
            <Select.Option value="2" children="Two" />
            <Select.Option value="3" children="Three" />
          </Select>
        </Field>

        <Field>
          <Field.Label children="Month" />
          <Select>
            <Select.Option value="1" children="One" />

            <Select.Option value="3" children="Three" />
          </Select>
        </Field>

        <Field>
          <Field.Label children="Year" />
          <Select>
            <Select.Option value="1" children="One" />

          pattern={/[!?@#]/}
          children="One special character (!?@#)"

    <Field direction="horizontal" className="pt-4">
