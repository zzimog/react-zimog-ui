# Input

Basic `<Input />` component to input data.

<Demo>
  <Field className="mx-auto max-w-md">
    <Field.Label htmlFor="demo-input" children="Lorem ipsum dolor sit" />
    <Input id="demo-input" />
    <Field.Description children="Lorem ipsum dolor sit amet consectetur adipisicing elit." />
  </Field>
</Demo>

```tsx
<Field>
  <Field.Label />
  <Input />
  <Field.Description />
</Field>
```

## Disabled

Use the `disabled` prop to disable the input.

<Demo>
  <Field className="mx-auto max-w-md">
    <Field.Label
      htmlFor="demo-input-disabled"
      children="Lorem ipsum dolor sit"
    />
    <Input id="demo-input-disabled" placeholder="Disabled input" disabled />
  </Field>
</Demo>

```tsx
<Field>
  <Field.Label />
  <Input disabled />
</Field>
```

## Invalid

Use `aria-invalid="true"` to set the input as invalid.

<Demo>
  <Field className="mx-auto max-w-md">
    <Field.Label
      htmlFor="demo-input-invalid"
      children="Lorem ipsum dolor sit"
    />
    <Input id="demo-input-invalid" aria-invalid="true" />
  </Field>
</Demo>

```tsx
<Field>
  <Field.Label />
  <Input aria-invalid="true" />
</Field>
```

## Group & Addons

Wrap your `<Input />` in a `<Input.Group />` component and add `<Input.Addon />` to show icons or others decorative elements in your field.

<Demo>
  <div className="mx-auto flex max-w-md flex-col gap-4">
    <Input.Group>
      <Input.Addon>context/</Input.Addon>
      <Input id="demo-input-1" placeholder="resource" />
    </Input.Group>
    <Input.Group>
      <Input.Addon>http://</Input.Addon>
      <Input id="demo-input-2" placeholder="your-subdomain" />
      <Input.Addon>.example.com</Input.Addon>
    </Input.Group>
    <Input.Group>
      <Input
        id="demo-input-3"
        placeholder="name.surname"
        className="text-right"
      />
      <Input.Addon>@example.com</Input.Addon>
    </Input.Group>
  </div>
</Demo>

```tsx
<Input.Group>
  <Input.Addon>context/</Input.Addon>
  <Input />
</Input.Group>

<Input.Group>
  <Input.Addon>http://</Input.Addon>
  <Input />
  <Input.Addon>.example.com</Input.Addon>
</Input.Group>

<Input.Group>
  <Input />
  <Input.Addon>@example.com</Input.Addon>
</Input.Group>
```
